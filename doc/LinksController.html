def create
        newLink = Link.new
        newLink.url = params[:link][:url]
        newLink.done = false
        newLink.description = params[:link][:description]
        newLink.save
        redirect_to link_list_path
end

def done
        link = Link.find_by("id = " + params[:id])

        if link
                link.done = true
                link.save
        end
end
